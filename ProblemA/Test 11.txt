"""
N, K
A[i]-x -> into a deque(sorted)
c = 1
while deque not empty:
	delete all elems less than c, stop when c >= x to save time
	input k elems, or whatever is left
	r += k (or whatever if left)
	delete all k elems
	c += 1
"""
T = int(input())
for t in range(1, T+1):
    n, k = map(int, input().split())
    a = [int(x) for x in input().split()]
    a.sort()
    c = 1
    r = 0
    while a:
        i = 0
        a = [x for x in a if x >= c]
        print(a)
        n = len(a)
        if n >= k:
            r += k
            print(r)
            del a[:k]
            c += 1
        else:
            r += (k-n-1)
            print(r)
            a.clear()
    print("Case #{}: {}".format(t, r))
